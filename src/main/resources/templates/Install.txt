---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: archive-${dupId}
  labels:
    app: archive-${dupId}
spec:
  selector:
    matchLabels:
      app: archive-${dupId}
  strategy:
    type: Recreate
  template:
    metadata:
      labels:
        app: archive-${dupId}
    spec:
      containers:
        - name: archive-${dupId}
          image: {{.Values.fileStorage.imageName}}:{{.Values.fileStorage.imageVersion}}
          ports:
            - containerPort: 80
              name: file-server
          volumeMounts:
            - mountPath: "/usr/share/nginx/html"
              name: archive-volume
            - name: nginx-config
              mountPath: /etc/nginx/conf.d/default.conf
              subPath: default.conf
          imagePullPolicy: Always
          resources: {}
          lifecycle:
            postStart:
              exec:
                command: ["bin/sh", "-c", "chmod 777 /usr/share/nginx/html"]
      volumes:
        - name: archive-volume
          persistentVolumeClaim:
            claimName: archive-${dupId}-claim
        - name: nginx-config
          configMap:
            name: nginx-archives-configmap
---
apiVersion: v1
kind: Service
metadata:
  labels:
    service: archive-${dupId}
  name: archive-${dupId}
spec:
  ports:
    - name: http
      port: 80
  selector:
    app: archive-${dupId}
status:
  loadBalancer: {}
---
kind: PersistentVolumeClaim
apiVersion: v1
metadata:
  name: archive-${dupId}-claim
spec:
  storageClassName: nfs
  accessModes:
    - ReadWriteMany
  resources:
    requests:
      storage: {{.Values.fileStorage.size}}
